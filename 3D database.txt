"A Multi-Detector, Ablation-Controlled, and Statistically Validated Evaluation of a Hybrid 3D Geometric–Radiometric Descriptor for Structural Anomaly Detection"

# 3D-FPFH-Int: Reproducible Research Package

This repository contains the complete implementation for the paper:

**"A Multi-Detector, Ablation-Controlled, and Statistically Validated Evaluation of a Hybrid 3D Geometric–Radiometric Descriptor for Structural Anomaly Detection"**

## Requirements

- Docker (recommended) or Python 3.10 with dependencies listed in `requirements.txt`
- At least 16 GB RAM for full synthetic dataset (500 clouds, 500k points each)

## Data

- **Synthetic dataset** is generated automatically by `synthetic/generator.py`. The generation parameters follow the specifications in the paper (500 clouds, crack widths 0.3–5.0 mm, Gaussian noise σ=0.5 mm, adversarial subsets with 40% contrast reduction and Perlin noise displacement).

- **Real-world data**: Anonymised subsets of the Berok Bridge and Lawang Sewu tunnel scans are provided in `data/berok_bridge/` and `data/lawang_sewu/` for verification purposes. These include point clouds in `.ply` format and corresponding ground truth masks (`.npy` files). The full datasets are not publicly available due to cultural heritage protection regulations. Researchers interested in accessing the complete datasets may contact the corresponding author; a data-sharing agreement can be arranged.

#!/usr/bin/env python3
"""
Unified entry point for running experiments.
"""

import argparse
import yaml
import importlib
import sys
import os
sys.path.insert(0, os.path.abspath(os.path.dirname(__file__)))

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument('--config', required=True, help='Path to YAML config file')
    parser.add_argument('--seed', type=int, default=42, help='Random seed')
    args = parser.parse_args()

    with open(args.config, 'r') as f:
        config = yaml.safe_load(f)

    experiment_type = config.get('experiment')
    if experiment_type is None:
        raise ValueError("Config file must specify 'experiment' field")

    module = importlib.import_module(f'{experiment_type}.run_{experiment_type}')
    module.run(config, args.seed)

if __name__ == '__main__':
    main()

"""
Utility functions: I/O, seeding, point cloud helpers.
"""

import numpy as np
import open3d as o3d
import random
import os

def set_seeds(seed=42):
    random.seed(seed)
    np.random.seed(seed)

def load_point_cloud(path, load_mask=False):
    pcd = o3d.io.read_point_cloud(path)
    points = np.asarray(pcd.points)
    intensities = np.asarray(pcd.colors)[:, 0] if pcd.colors else None
    mask = None
    if load_mask:
        mask_path = path.replace('.ply', '_mask.npy')
        if os.path.exists(mask_path):
            mask = np.load(mask_path)
    return points, intensities, pcd, mask

def save_point_cloud(points, intensities, path):
    pcd = o3d.geometry.PointCloud()
    pcd.points = o3d.utility.Vector3dVector(points)
    if intensities is not None:
        colors = np.stack([intensities, intensities, intensities], axis=1)
        pcd.colors = o3d.utility.Vector3dVector(colors)
    o3d.io.write_point_cloud(path, pcd)

"""
Enhanced FPFH with vectorised operations and 30 bins per angular feature.
"""

import numpy as np
from scipy.spatial import KDTree

def fpfh_descriptor(points, normals, radius=0.05, bins=30):
    N = points.shape[0]
    tree = KDTree(points)
    neighbors = tree.query_ball_point(points, radius)

    desc = np.zeros((N, bins * 3), dtype=np.float32)

    for i in range(N):
        idx = neighbors[i]
        if len(idx) < 2:
            continue
        p_i = points[i]
        n_i = normals[i]
        pts_j = points[idx[1:]]
        n_j = normals[idx[1:]]
        delta = pts_j - p_i

        u = n_i
        cross_v = np.cross(delta, u)
        norm_v = np.linalg.norm(cross_v, axis=1, keepdims=True)
        norm_v[norm_v == 0] = 1
        v = cross_v / norm_v
        w = np.cross(u, v)

        alpha = np.arctan2(np.sum(w * n_j, axis=1), np.sum(u * n_j, axis=1))
        phi = np.sum(v * n_j, axis=1)
        theta = np.arctan2(np.sum(w * delta, axis=1), np.sum(u * delta, axis=1))

        hist_alpha, _ = np.histogram(alpha, bins=bins, range=(-np.pi, np.pi), density=False)
        hist_phi, _ = np.histogram(phi, bins=bins, range=(-1, 1), density=False)
        hist_theta, _ = np.histogram(theta, bins=bins, range=(-np.pi, np.pi), density=False)
        desc[i] = np.concatenate([hist_alpha, hist_phi, hist_theta])

    return desc

"""
Local 3D intensity histogram with range computed from training data.
"""

import numpy as np
from scipy.spatial import KDTree

def intensity_histogram(points, intensities, radius=0.05, bins=30, range_min=None, range_max=None):
    N = points.shape[0]
    tree = KDTree(points)
    neighbors = tree.query_ball_point(points, radius)

    if range_min is None or range_max is None:
        all_diffs = []
        for i in range(N):
            idx = neighbors[i]
            if len(idx) < 2:
                continue
            diffs = intensities[idx[1:]] - intensities[i]
            all_diffs.extend(diffs)
        range_min = min(all_diffs)
        range_max = max(all_diffs)
        if range_min == range_max:
            range_max = range_min + 1

    hist = np.zeros((N, bins), dtype=np.float32)
    for i in range(N):
        idx = neighbors[i]
        if len(idx) < 2:
            continue
        diffs = intensities[idx[1:]] - intensities[i]
        h, _ = np.histogram(diffs, bins=bins, range=(range_min, range_max), density=False)
        hist[i] = h
    return hist, (range_min, range_max)

"""
3D-FPFH-Int descriptor with two-stage normalisation and global scaling.
"""

import numpy as np
from .fpfh import fpfh_descriptor
from .intensity_hist import intensity_histogram

def compute_descriptor(points, intensities, normals, radius=0.05, bins_fpfh=30, bins_int=30, int_range=None):
    f_g = fpfh_descriptor(points, normals, radius=radius, bins=bins_fpfh)
    f_i, int_range = intensity_histogram(points, intensities, radius=radius, bins=bins_int,
                                         range_min=int_range[0] if int_range else None,
                                         range_max=int_range[1] if int_range else None)
    return np.hstack([f_g, f_i]), int_range

def normalise_descriptors(desc_train, desc_test):
    # L2 normalise each row
    norms_train = np.linalg.norm(desc_train, axis=1, keepdims=True)
    norms_train[norms_train == 0] = 1
    desc_train_norm = desc_train / norms_train

    norms_test = np.linalg.norm(desc_test, axis=1, keepdims=True)
    norms_test[norms_test == 0] = 1
    desc_test_norm = desc_test / norms_test

    # Compute scaling factor λ = std(geometric norms) / std(radiometric norms)
    norms_g_train = np.linalg.norm(desc_train_norm[:, :90], axis=1)
    norms_i_train = np.linalg.norm(desc_train_norm[:, 90:], axis=1)
    std_g = np.std(norms_g_train)
    std_i = np.std(norms_i_train)
    λ = 1.0 if std_i == 0 else std_g / std_i

    # Apply scaling to radiometric part only
    desc_train_scaled = np.copy(desc_train_norm)
    desc_train_scaled[:, 90:] *= λ
    desc_test_scaled = np.copy(desc_test_norm)
    desc_test_scaled[:, 90:] *= λ

    return desc_train_scaled, desc_test_scaled, λ

"""
Morphological opening on point clouds using radius graph.
"""

import numpy as np
from scipy.spatial import KDTree

def morphological_opening(points, labels, radius):
    tree = KDTree(points)
    neighbors = tree.query_ball_point(points, radius)

    # Erosion
    eroded = np.zeros_like(labels)
    for i, idx in enumerate(neighbors):
        if len(idx) > 0:
            eroded[i] = int(np.any(labels[idx]))

    # Dilation
    dilated = np.zeros_like(labels)
    for i, idx in enumerate(neighbors):
        if len(idx) > 0:
            dilated[i] = int(np.any(eroded[idx]))
    return dilated

"""
Localization error computation.
"""

import numpy as np
from scipy.spatial import KDTree

def compute_localization_error(pred_mask, gt_centerline_points, points):
    if np.sum(pred_mask) == 0:
        return np.nan
    pred_points = points[pred_mask]
    tree = KDTree(gt_centerline_points)
    distances, _ = tree.query(pred_points)
    return np.mean(distances)

def compute_false_positive_rate(pred_mask, gt_mask):
    fp = np.sum((pred_mask == 1) & (gt_mask == 0))
    total = len(gt_mask)
    return fp / total * 1000  # per 1000 points

"""
PatchCore‑style detector with coreset, Youden threshold optimisation, and morphological post‑processing.
"""

import numpy as np
from sklearn.neighbors import NearestNeighbors
from sklearn.metrics import roc_curve
from .base import BaseDetector
from src.morphology import morphological_opening

def greedy_coreset(features, budget):
    n = features.shape[0]
    if n <= budget:
        return np.arange(n)
    selected = [np.random.randint(n)]
    remaining = set(range(n)) - set(selected)
    while len(selected) < budget:
        dist = np.min([np.linalg.norm(features[list(remaining)] - features[s], axis=1) for s in selected], axis=0)
        farthest = list(remaining)[np.argmax(dist)]
        selected.append(farthest)
        remaining.remove(farthest)
    return np.array(selected)

class PatchCore(BaseDetector):
    def __init__(self, coreset_fraction=0.1, threshold=None, morph_radius=0.05, seed=42):
        self.coreset_fraction = coreset_fraction
        self.threshold = threshold
        self.morph_radius = morph_radius
        self.seed = seed
        np.random.seed(seed)
        self.memory_bank = None
        self.nn = None

    def fit(self, features):
        n = features.shape[0]
        budget = max(1, int(n * self.coreset_fraction))
        idx = greedy_coreset(features, budget)
        self.memory_bank = features[idx]
        self.nn = NearestNeighbors(n_neighbors=1, metric='euclidean').fit(self.memory_bank)

    def predict(self, features):
        if self.nn is None:
            raise RuntimeError("Model not fitted.")
        distances, _ = self.nn.kneighbors(features)
        return distances.flatten()

    def optimize_threshold(self, features_val, labels_val):
        scores = self.predict(features_val)
        fpr, tpr, thresholds = roc_curve(labels_val, scores)
        J = tpr - fpr
        best_idx = np.argmax(J)
        self.threshold = thresholds[best_idx]
        return self.threshold

    def threshold_predict(self, features, points=None):
        scores = self.predict(features)
        pred = (scores > self.threshold).astype(int)
        if points is not None and self.morph_radius > 0:
            pred = morphological_opening(points, pred, self.morph_radius)
        return pred

from sklearn.ensemble import IsolationForest

class IsolationForestDetector(BaseDetector):
    def __init__(self, n_estimators=200, max_samples=256, contamination='auto', random_state=0):

            n_estimators=n_estimators,

def arch_points(span=8.0, rise=2.0, thickness=0.45, width=4.0, num_points=500000, seed=42):
    np.random.seed(seed)
    R_int = (span**2 + 4*rise**2) / (8*rise)
    center_y = rise - R_int
    R_ext = R_int + thickness

    n_int = num_points // 2
    n_ext = num_points - n_int

    # Intrados
    theta_int = np.random.uniform(np.pi/2, 3*np.pi/2, n_int)
    x_int = R_int * np.cos(theta_int)
    y_int = center_y + R_int * np.sin(theta_int)
    z_int = np.random.uniform(-width/2, width/2, n_int)
    points_int = np.stack([x_int, y_int, z_int], axis=1)
    normals_int = -np.stack([np.cos(theta_int), np.sin(theta_int), np.zeros_like(theta_int)], axis=1)
    normals_int = normals_int / np.linalg.norm(normals_int, axis=1, keepdims=True)

    # Extrados
    theta_ext = np.random.uniform(np.pi/2, 3*np.pi/2, n_ext)
    x_ext = R_ext * np.cos(theta_ext)
    y_ext = center_y + R_ext * np.sin(theta_ext)
    z_ext = np.random.uniform(-width/2, width/2, n_ext)
    points_ext = np.stack([x_ext, y_ext, z_ext], axis=1)
    normals_ext = np.stack([np.cos(theta_ext), np.sin(theta_ext), np.zeros_like(theta_ext)], axis=1)
    normals_ext = normals_ext / np.linalg.norm(normals_ext, axis=1, keepdims=True)

    points = np.vstack([points_int, points_ext])
    normals = np.vstack([normals_int, normals_ext])
    idx = np.random.permutation(num_points)
    return points[idx], normals[idx]

def add_crack(points, normals, width, position=0.0, seed=None):
    np.random.seed(seed)
    mask = np.abs(points[:,0] - position) < width/2
    displacement = np.random.uniform(-width/4, width/4, size=points.shape)
    displacement[mask] *= 2
    points_new = points + displacement
    return points_new, normals, mask

import numpy as np

def simulate_intensity(points, normals, crack_mask=None, base_reflectivity=0.4,
                       incidence_angle_weight=0.3, crack_attenuation=0.6, noise_sigma=0.05):
    scanner_dir = np.array([1.0, 0.0, 0.0])
    cos_theta = np.abs(np.dot(normals, scanner_dir))
    cos_theta = np.clip(cos_theta, 0.1, 1.0)

    intensity = base_reflectivity * cos_theta
    if crack_mask is not None:
        intensity[crack_mask] *= crack_attenuation

    intensity += np.random.normal(0, noise_sigma, len(points))
    return np.clip(intensity, 0.0, 1.0)

def load_synthetic_data(data_dir, split, load_normals=True, load_masks=False):
    if split == 'train':

    elif split == 'test':
        prefix = 'cracked'
    elif split == 'weak':
        prefix = 'weak'
    elif split == 'complex':

    elif split == 'val_normal':

    elif split == 'val_cracked':
        prefix = 'cracked'  # we'll use a subset of cracked for validation
    else:
        raise ValueError(f"Unknown split: {split}")

    files = [os.path.join(data_dir, f) for f in os.listdir(data_dir) if f.startswith(prefix) and f.endswith('.ply')]
    files.sort()

            else:
                normals_list.append(compute_normals(pts, radius))

            masks.append(mask)

def compute_features_for_clouds(clouds, intensities, normals_list, radius, bins_fpfh, bins_int, variant='hybrid', int_range=None):
    all_feat = []

def downsample_points(points, factor):
    n = len(points)
    keep = int(n * factor)
    idx = np.random.choice(n, keep, replace=False)
    return points[idx], idx

def add_gaussian_noise(intensities, sigma_percent):
    sigma = sigma_percent / 100.0
    noise = np.random.normal(0, sigma, len(intensities))
    return np.clip(intensities + noise, 0, 1)

def apply_intensity_drift(intensities, drift_dn, original_range=255):
    dn = intensities * original_range
    dn_shifted = dn + drift_dn
    dn_shifted = np.clip(dn_shifted, 0, original_range)
    return dn_shifted / original_range

def reduce_contrast(intensities, factor=0.4):
    mean_int = np.mean(intensities)
    return mean_int + factor * (intensities - mean_int)

def apply_registration_error(points, trans_sigma_mm, rot_sigma_deg):
    trans = np.random.normal(0, trans_sigma_mm/1000.0, 3)
    rot_rad = np.radians(rot_sigma_deg)
    angle = np.random.normal(0, rot_rad)
    axis = np.random.randn(3)
    axis = axis / np.linalg.norm(axis)
    K = np.array([[0, -axis[2], axis[1]],
                  [axis[2], 0, -axis[0]],
                  [-axis[1], axis[0], 0]])
    R = np.eye(3) + np.sin(angle) * K + (1 - np.cos(angle)) * K @ K
    points_rot = points @ R.T
    points_trans = points_rot + trans
    return points_trans

def benchmark_descriptor(points, intensities, normals, func, radius, bins_fpfh, bins_int, n_runs=5):
    times = []
    mems = []
    for _ in range(n_runs):
        tracemalloc.start()
        start = time.time()
        _ = func(points, intensities, normals, radius, bins_fpfh, bins_int)
        end = time.time()
        current, peak = tracemalloc.get_traced_memory()
        tracemalloc.stop()
        times.append(end - start)
        mems.append(peak / (1024**2))
    return np.mean(times), np.mean(mems)

def main():
    with open('configs/benchmark.yaml', 'r') as f:
        config = yaml.safe_load(f)

    # Load a sample cloud (first intact from synthetic)

def bca_ci(data, statistic, n_resamples=1000, alpha=0.05, seed=42):
    np.random.seed(seed)
    result = skb.bootstrap(data, statistic, n_boot=n_resamples, alpha=alpha, method='bca')

"""
Cohen's d with confidence intervals via noncentral t-distribution.
"""

import numpy as np
from scipy import stats

def cohens_d(x, y):
    nx = len(x)
    ny = len(y)
    dof = nx + ny - 2
    pooled_std = np.sqrt(((nx-1)*np.std(x, ddof=1)**2 + (ny-1)*np.std(y, ddof=1)**2) / dof)
    d = (np.mean(x) - np.mean(y)) / pooled_std
    return d

def cohens_d_ci(x, y, alpha=0.05):
    d = cohens_d(x, y)
    nx, ny = len(x), len(y)
    dof = nx + ny - 2
    ncp = d * np.sqrt((nx*ny)/(nx+ny))
    t_low = stats.t.ppf(alpha/2, dof)
    t_high = stats.t.ppf(1-alpha/2, dof)
    ncp_ci = (ncp - t_high, ncp - t_low)
    factor = np.sqrt((nx+ny)/(nx*ny))
    d_low = ncp_ci[0] * factor
    d_high = ncp_ci[1] * factor
    return d, (d_low, d_high)

"""
Two‑way ANOVA with assumption checks and effect sizes, including CIs for η².
"""

import numpy as np
import pandas as pd
from scipy import stats
import statsmodels.api as sm
from statsmodels.formula.api import ols

def eta_squared_ci(eta_sq, df1, df2, alpha=0.05):
    f = eta_sq / (1 - eta_sq) * df2 / df1
    ncp_low = stats.ncf.ppf(alpha/2, df1, df2, f*df1)
    ncp_high = stats.ncf.ppf(1-alpha/2, df1, df2, f*df1)
    eta_low = ncp_low / (ncp_low + df2)
    eta_high = ncp_high / (ncp_high + df2)
    return eta_low, eta_high

def two_way_anova(data, dependent, factors):
    groups = data.groupby(factors)[dependent].apply(list)
    for name, group in groups.items():
        _, p = stats.shapiro(group)
        if p < 0.05:
            print(f"Shapiro-Wilk p={p:.4f} for group {name}")

    group_vals = [data[(data[factors[0]]==f1) & (data[factors[1]]==f2)][dependent].values
                  for f1 in data[factors[0]].unique() for f2 in data[factors[1]].unique()]
    stat, p_levene = stats.levene(*group_vals)
    if p_levene < 0.05:
        print(f"Levene p={p_levene:.4f}")

    formula = f"{dependent} ~ C({factors[0]}) + C({factors[1]}) + C({factors[0]}):C({factors[1]})"
    model = ols(formula, data=data).fit()
    aov = sm.stats.anova_lm(model, typ=2)

    ss_effect = aov['sum_sq'][:-1]
    ss_total = ss_effect.sum() + aov['sum_sq']['Residual']
    eta_sq = ss_effect / ss_total

    result = {
        'SS': aov['sum_sq'].to_dict(),
        'df': aov['df'].to_dict(),
        'F': aov['F'].to_dict(),
        'p': aov['PR(>F)'].to_dict(),
        'η²': eta_sq.to_dict(),
        'η²_CI': {}
    }
    for key in eta_sq.index:
        if key != 'Residual':
            df1 = aov['df'][key]
            df2 = aov['df']['Residual']
            low, high = eta_squared_ci(eta_sq[key], df1, df2, alpha=0.05)
            result['η²_CI'][key] = (low, high)
    return result

"""
Post‑hoc power analysis for two‑way ANOVA interaction.
"""

from scipy import stats

def compute_power(eta_sq, df1, df2, alpha=0.05):
    f = eta_sq / (1 - eta_sq) * df2 / df1
    ncp = f * df1
    f_crit = stats.f.ppf(1-alpha, df1, df2)
    power = 1 - stats.ncf.cdf(f_crit, df1, df2, ncp)
    return power
